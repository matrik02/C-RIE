var SPREADSHEET_ID = "18sQkH8sGe82SmUtFCpqv8PQrW59RwwM7KJpzgRBoFxM"; 
var ss = SpreadsheetApp.openById(SPREADSHEET_ID);
// =========================================================================
function doGet(e) {
  var action = "";
  if (e && e.parameter && e.parameter.action) {
    action = e.parameter.action;
  }


// BUKA SATU SAJA BLOK TRY UNTUK SEMUA FUNGSI
  try {

// ==========================================
    // 1. FUNGSI DOWNLOAD TEMPLATE (KALIS MULTI-ACCOUNT)

      var files = folder.getFiles();
      var latestUrl = "";

      // Cari fail di dalam folder tersebut
      while (files.hasNext()) {
        var file = files.next();
        var fileName = file.getName();
        
        // Pastikan ia BUKAN fail yang dah dipadam atau diarkib
        if (!fileName.startsWith("[ARKIB]") && !fileName.startsWith("[DIPADAM]")) {
          latestUrl = "https://drive.google.com/uc?export=download&id=" + file.getId();
          break; // Berhenti mencari selepas jumpa fail aktif yang pertama
        }
      }

      // 🌟 KEMASKINI PENTING: Pulangkan sebagai JSON, BUKAN HTML 🌟
      if (latestUrl !== "") {
        return ContentService.createTextOutput(JSON.stringify({
          "status": "BERJAYA",
          "url": latestUrl
        })).setMimeType(ContentService.MimeType.JSON);
      } else {
        return ContentService.createTextOutput(JSON.stringify({
          "status": "GAGAL",
          "mesej": "Tiada fail template ditemui."
        })).setMimeType(ContentService.MimeType.JSON);
      }
    }

    // --- A. AMBIL DATA INOVASI ---
    if (action === "ambilData") {
      var sheetInovasi = ss.getSheetByName("INOVASI");
      if (!sheetInovasi) return ContentService.createTextOutput(JSON.stringify({"status": "GAGAL", "mesej": "Tab 'INOVASI' tidak dijumpai."})).setMimeType(ContentService.MimeType.JSON);
      
      // 🌟 Guna getDisplayValues() supaya ia baca nombor formula dengan tepat
      var rowsInovasi = sheetInovasi.getDataRange().getDisplayValues(); 
      if (rowsInovasi.length <= 1) return ContentService.createTextOutput(JSON.stringify({"status": "BERJAYA", "data": []})).setMimeType(ContentService.MimeType.JSON);
      
      rowsInovasi.shift(); // Buang header
      
      // Filter data kosong
      var dataBersih = rowsInovasi
        .filter(function(r) { return r[1] && String(r[1]).trim() !== ""; })
        .map(function(r) { // ❌ BUANG 'index' di sini, kita tak perlukannya lagi
          
          // Jika r[9] tiada (rekod lama), kita letakkan default "MENUNGGU PENGESAHAN"
          var status = r[9] ? String(r[9]).trim() : "MENUNGGU PENGESAHAN";
          
          // Ambil link sijil dari Lajur K (Index 10)
          var linkSijil = r[10] ? String(r[10]).trim() : "";

          // ✅ AMARAN: Jangan guna bilAuto. Mesti guna r[0] supaya nombor BIL tepat dengan Sheet!
          return [
            r[0], // 🌟 MESTI r[0] (Ini adalah BIL sebenar di dalam Sheet)
            r[1], r[2], r[3], r[4], 
            r[5], r[6], r[7], r[8], status,
            linkSijil // Wajib hantar lajur ke-11 untuk paparan butang Download
          ];
        });
        
      return ContentService.createTextOutput(JSON.stringify({"status": "BERJAYA", "data": dataBersih})).setMimeType(ContentService.MimeType.JSON);
    }

 

    else if (action == "padamData") {
      return fungsiPadamRekod(e.parameter.id);
    }

    // --- B. AMBIL DATA KPI ---

      return ContentService.createTextOutput(JSON.stringify({
        "status": "BERJAYA", "data": senaraiPending
      })).setMimeType(ContentService.MimeType.JSON);

             tarikhFormat = String(rawDate).trim();

// --- AMBIL DATA DROPDOWN DINAMIK ---
    else if (action === "ambilPilihan") {

      // Fungsi utiliti untuk cari lajur berdasarkan tajuk baris pertama
      function getColData(data, headerName) {
        var colIndex = -1;
        if(data.length === 0) return [];
        for (var i = 0; i < data[0].length; i++) {
          if (String(data[0][i]).toUpperCase().trim() === headerName.toUpperCase()) {
            colIndex = i; break;
          }
        }
        if (colIndex === -1) return [];
        var colData = [];
        for (var j = 1; j < data.length; j++) {
          if (String(data[j][colIndex]).trim() !== "") {
            colData.push(String(data[j][colIndex]).trim());
          }
        }
        return [...new Set(colData)]; // Buang duplikasi
      }

      return ContentService.createTextOutput(JSON.stringify({
        "status": "BERJAYA", 
        "jabatan": getColData(dataSistem, "JABATAN"),
        "tahapSebaran": getColData(dataSistem, "TAHAP SEBARAN"),
        "bahasa": getColData(dataSistem, "BAHASA"),
        "laporanKpi": getColData(dataKpi, "SUKUAN")
      })).setMimeType(ContentService.MimeType.JSON);
    }

  // --- AMBIL SENARAI JABATAN DARI SHEET SISTEM ---

        var jab = String(dataSistem[i][0]).trim();
        if (jab !== "") {
          senaraiJabatan.push(jab);
        }

      // Buang jabatan yang berulang supaya unik
      var jabatanUnik = [...new Set(senaraiJabatan)];

function dapatkanSenaraiPlatform() {
  try {
    var sheet = ss.getSheetByName("SISTEM");
    if (!sheet) return ["RALAT: Tab SISTEM tidak dijumpai"];
    
    var data = sheet.getDataRange().getValues(); // Ambil semua data
    if (data.length < 2) return ["Tiada Data"];

    var headers = data[0];
    var colIndex = -1;

    // Cari lajur "JENIS PLATFORM" secara dinamik (Kalis ralat nombor lajur)
    for (var i = 0; i < headers.length; i++) {
      if (headers[i].toString().toUpperCase().trim() === "JENIS PLATFORM") {
        colIndex = i;
        break;
      }
    }

    if (colIndex === -1) return ["RALAT: Lajur 'JENIS PLATFORM' tidak ditemui"];

    var senarai = [];
    for (var j = 1; j < data.length; j++) {
      var cellValue = data[j][colIndex];
      if (cellValue && String(cellValue).trim() !== "") {
        var bersih = String(cellValue).trim();
        if (senarai.indexOf(bersih) === -1) {
          senarai.push(bersih);
        }
      }
    }
    
    return senarai.length > 0 ? senarai.sort() : ["Tiada pilihan ditemui"];
  } catch (e) {
    return ["RALAT SERVER: " + e.message];
  }
}

function doPost(e) {
// 1. Sediakan pembolehubah 'output' paling awal
  var output = ContentService.createTextOutput().setMimeType(ContentService.MimeType.JSON);
// 2. Tangkap dan parse data yang dihantar dari fail HTML
  var data;
  var action;

  try {
    // Jika data dihantar dalam format JSON
    data = JSON.parse(e.postData.contents);
    action = data.action; 
  } catch (error) {
    // Jika data dihantar menggunakan Form biasa
    data = e.parameter || {};
    action = data.action;
  }
    // Pastikan 'data' tidak undefined sebelum meneruskan
    if (!data) {
        return ContentService.createTextOutput(JSON.stringify({"status": "GAGAL", "mesej": "Tiada data diterima."})).setMimeType(ContentService.MimeType.JSON);
    }

// ==========================================
  // 2. PENGURUSAN ARAHAN (LOGIK IF-ELSE)
  // ==========================================

  // --- A. KEMASKINI TEMPLATE BORANG ---
  if (action === "kemaskiniTemplateBorang") {
    // Pastikan function kemaskiniTemplateBorang(data) wujud di skrip anda
    var hasil = kemaskiniTemplateBorang(data); 
    return output.setContent(JSON.stringify(hasil));
  }

  // --- B. UPLOAD TEMPLATE SENARAI SEMAK KPI & PADAM FAIL LAMA ---
  else if (action === "uploadTemplate") {
    try {
      // ID Folder Kamus KPI anda
      var folderId = "1NYqDHUyaz1HTnbJh4_6kh1Axgu60WaoC"; 
      var folder = DriveApp.getFolderById(folderId);
      
      // 1. CARI DAN PADAM (MASUK TONG SAMPAH) SEMUA FAIL LAMA DALAM FOLDER
      var files = folder.getFiles();
      while (files.hasNext()) {
        var oldFile = files.next();
        try {
          oldFile.setTrashed(true); // Buang fail lama ke tong sampah
        } catch (err) {
          // Jika dihalang (Access denied sebab bukan owner), kita ubah nama fail
          oldFile.setName("[DIPADAM] " + oldFile.getName());
        }
      }

      // 2. MUAT NAIK FAIL BAHARU
      // Tidak perlu JSON.parse lagi kerana ia sudah dilakukan di bahagian atas
      var failBase64 = Utilities.base64Decode(data.base64);
      var blob = Utilities.newBlob(failBase64, data.mimeType, data.fileName);
      var newFile = folder.createFile(blob);

      // Beri keizinan supaya fail boleh dilihat oleh sesiapa sahaja melalui link
      //newFile.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);

      // Pulangkan respon BERJAYA ke HTML
      return output.setContent(JSON.stringify({ status: "BERJAYA", url: newFile.getUrl() }));
        
    } catch (error) {
      // Jika ralat berlaku

  }

// Fungsi OPTIONS untuk mengelakkan ralat CORS (kekalkan di atas atau luar)
  if (e && e.postData && e.postData.type === "application/x-www-form-urlencoded" && Object.keys(e.parameter).length === 0) {

  }
  try {
    var payload = {};
    if (e && e.postData && e.postData.contents) {

    } else { payload = e.parameter || {}; }

    var ss = SpreadsheetApp.openById(SPREADSHEET_ID);

      var produk    = payload.produk_inovasi || payload.PRODUK || payload.produk || "";

      var tempat_input = payload.TEMPAT || payload.tempat_pertandingan || "";
      var tarikh = (tarikh_input !== "" && tempat_input !== "") ? tarikh_input + " | " + tempat_input : (tarikh_input || tempat_input || payload.TARIKHTEMPATPERTANDINGAN || "");

      var statusInovasi = payload.status || payload.STATUS || "MENUNGGU PENGESAHAN";

      if (innovator === "" && produk === "") return output.setContent(JSON.stringify({"status": "GAGAL", "mesej": "Tiada maklumat sah."}));

            }

        } catch (e) {
            return output.setContent(JSON.stringify({"status": "GAGAL", "mesej": "Ralat memproses CSV: " + e.message}));
        }
    }

    return output.setContent(JSON.stringify({"status": "GAGAL", "mesej": "Arahan tidak dikenali."}));

  } catch (error) {
    return output.setContent(JSON.stringify({"status": "GAGAL", "mesej": "Ralat Server: " + error.message}));
  }
}

function fungsiPadamRekod(id) {
  // PASTIKAN NAMA SHEET BETUL. Contoh: "INOVASI"
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("INOVASI"); 

  // 🌟 TAMBAHAN KESELAMATAN: Semak jika tab tidak wujud
  if (!sheet) {
    return ContentService.createTextOutput(JSON.stringify({
      status: "GAGAL",
      mesej: "Tab 'INOVASI' tidak dijumpai dalam database."
    })).setMimeType(ContentService.MimeType.JSON);
  }

  var data = sheet.getDataRange().getValues();
  var barisDitemui = -1;
  
  // Mula loop dari baris 2 (index 1) kerana baris 1 adalah tajuk (Header)
  for (var i = 1; i < data.length; i++) {
    // data[i][0] bermaksud Kolum A (BIL). Jika ID anda di kolum lain, tukar index 0.
    if (data[i][0] == id) { 
      barisDitemui = i + 1; // +1 kerana index array mula dari 0, baris sheet mula dari 1
      break;
    }
  }
  
  if (barisDitemui !== -1) {
    sheet.deleteRow(barisDitemui); // Padam baris tersebut
    
    // ==============================================================
    // 🌟 KOD BARU: SUSUN SEMULA NOMBOR BIL SECARA FIZIKAL DI SHEET
    // ==============================================================
    var barisTerakhir = sheet.getLastRow();
    
    // Semak jika masih ada data (lebih dari baris 1/tajuk)
    if (barisTerakhir > 1) {
      var jumlahData = barisTerakhir - 1;
      var susunanNomborBaru = [];
      
      // Buat senarai nombor berturutan: 1, 2, 3, 4....
      for (var n = 1; n <= jumlahData; n++) {
        susunanNomborBaru.push([n]); 
      }
      
      // Masukkan senarai nombor baru ini terus ke Kolum A bermula dari Baris 2
      sheet.getRange(2, 1, susunanNomborBaru.length, 1).setValues(susunanNomborBaru);
    }
    // ==============================================================

    return ContentService.createTextOutput(JSON.stringify({
      status: "BERJAYA",
      mesej: "Rekod berjaya dipadam & nombor BIL telah disusun semula."
    })).setMimeType(ContentService.MimeType.JSON);
    
  } else {
    return ContentService.createTextOutput(JSON.stringify({
      status: "GAGAL",
      mesej: "ID Rekod tidak dijumpai di dalam database."
    })).setMimeType(ContentService.MimeType.JSON);
  }
}

function kemaskiniRekod(data, ss) {
  // Pastikan variabel NAMA_SHEET (contoh: "KPI") sudah dideklarasikan secara global
  const sheet = ss.getSheetByName(NAMA_SHEET); 
  
  // Semakan keselamatan: Jika nama sheet salah/ubah
  if (!sheet) {
    return jsonResponse({ status: "RALAT", mesej: `Sheet ${NAMA_SHEET} tidak dijumpai.` });
  }

  // ==========================================
  // SEMAKAN WAJIB (VALIDATION)
  // ==========================================
  // Pastikan namaProduk ada dan bukan sekadar ruang kosong
  if (!data.namaProduk || data.namaProduk.toString().trim() === "") {
    return jsonResponse({ status: "GAGAL", mesej: "Nama Produk adalah wajib diisi dan tidak boleh kosong." });
  }
  
  // Pastikan ketua ada dan bukan sekadar ruang kosong
  if (!data.ketua || data.ketua.toString().trim() === "") {
    return jsonResponse({ status: "GAGAL", mesej: "Nama Ketua adalah wajib diisi dan tidak boleh kosong." });
  }
  // ==========================================

  const idBil = data.id;
  const semuaData = sheet.getDataRange().getValues();
  
  for (let i = 1; i < semuaData.length; i++) {
    if (String(semuaData[i][0]) === String(idBil)) {
      const rowSheet = i + 1;
      
      // Kemaskini kolum. 
      // Untuk medan yang wajib (Nama Produk & Ketua), kita terus set nilainya kerana sudah disahkan tidak kosong di atas.
      sheet.getRange(rowSheet, 2).setValue(data.namaProduk);   // Col B (Wajib)
      sheet.getRange(rowSheet, 3).setValue(data.ketua);        // Col C (Wajib)
      
      // Untuk medan lain yang tidak wajib, kita kekalkan semakan !== undefined 
      // supaya jika pengguna sengaja padam (hantar ""), ia akan dikosongkan di Sheet.
      if(data.telefon !== undefined) sheet.getRange(rowSheet, 4).setValue(data.telefon);         // Col D
      if(data.email !== undefined) sheet.getRange(rowSheet, 5).setValue(data.email);             // Col E
      if(data.ahli !== undefined) sheet.getRange(rowSheet, 6).setValue(data.ahli);               // Col F
      if(data.jabatan !== undefined) sheet.getRange(rowSheet, 7).setValue(data.jabatan);         // Col G
      if(data.aplikasi !== undefined) sheet.getRange(rowSheet, 8).setValue(data.aplikasi);       // Col H
      if(data.jenis !== undefined) sheet.getRange(rowSheet, 9).setValue(data.jenis);             // Col I
      if(data.pendaftaran !== undefined) sheet.getRange(rowSheet, 10).setValue(data.pendaftaran);// Col J
      if(data.tahun !== undefined) sheet.getRange(rowSheet, 11).setValue(data.tahun);            // Col K
      if(data.url !== undefined) sheet.getRange(rowSheet, 12).setValue(data.url);                // Col L
      if(data.sukuan !== undefined) sheet.getRange(rowSheet, 13).setValue(data.sukuan);          // Col M
      if(data.status !== undefined) sheet.getRange(rowSheet, 14).setValue(data.status);          // Col N
      
      return jsonResponse({ status: "BERJAYA", mesej: `Rekod BIL ${idBil} berjaya dikemaskini.` });
    }
  }
  
  return jsonResponse({ status: "GAGAL", mesej: "Rekod tidak dijumpai untuk dikemaskini." });
}

function muatNaikSijil(data, ss) {
  const sheet = ss.getSheetByName(NAMA_SHEET);
  const idBil = data.id;
  const failBase64 = data.failBase64;     
  const namaFail = data.namaFail;   
  const mimeType = data.mimeType; 
  // 👇 TAMBAH INI: Ambil URL lama dari data yang dihantar
  const urlSijilLama = data.urlSijilLama || "";
  const semuaData = sheet.getDataRange().getValues();
  let rowSheet = -1;
  let tahunKPI = new Date().getFullYear().toString(); // Default jika tahun tak dijumpai

  // 1. CARI REKOD DAN DAPATKAN TAHUN KPI
  for (let i = 1; i < semuaData.length; i++) {
    if (String(semuaData[i][0]) === String(idBil)) {
      rowSheet = i + 1;
      // Ambil data tahun dari Lajur K (Index 10)
      if (semuaData[i][10] !== "") {
        tahunKPI = String(semuaData[i][10]).trim(); 
      }
      break;
    }
  }

  // Jika rekod tak wujud dalam sheet, hentikan proses
  if (rowSheet === -1) {
    return jsonResponse({ status: "GAGAL", mesej: "Rekod tidak dijumpai di pangkalan data." });
  }

// ==========================================
  // 🌟 TAMBAH INI: LOGIK PADAM FAIL LAMA
  // ==========================================
  if (urlSijilLama !== "") {
    try {
      // Ekstrak ID Fail dari pautan (URL) Google Drive lama
      const padananID = urlSijilLama.match(/[-\w]{25,}/);
      
      if (padananID && padananID[0]) {
        const idFailLama = padananID[0];
        // Pindahkan fail lama ke dalam Trash (Tong Sampah)
        DriveApp.getFileById(idFailLama).setTrashed(true);
      }
    } catch (e) {
      // Abaikan jika ralat (mungkin fail dah terpadam manual)
      console.log("Gagal memadam fail lama: " + e.message);
    }
  }
  // ==========================================


  // 2. PENGURUSAN FOLDER DINAMIK
  const folderUtama = DriveApp.getFolderById(FOLDER_ID_SIJILKPI);
  let folderTahun;
  
  // Semak jika folder tahun (cth: "2025") sudah wujud dalam folder utama
  const subFolders = folderUtama.getFoldersByName(tahunKPI);
  if (subFolders.hasNext()) {
    folderTahun = subFolders.next(); // Guna folder yang dah ada
  } else {
    folderTahun = folderUtama.createFolder(tahunKPI); // Bina folder tahun baharu
  }
  
// 3. 🌟 PENYELESAIAN RALAT "COULD NOT DECODE STRING" 🌟
  // Sistem akan automatik buang header 'data:image/xxx;base64,' jika ia wujud
  let teksBersih = failBase64;
  if (failBase64.indexOf(",") !== -1) {
    teksBersih = failBase64.split(",")[1]; 
  }
  
  // Convert teks bersih menjadi fail fizikal
  const decodedData = Utilities.base64Decode(teksBersih); 
  const blob = Utilities.newBlob(decodedData, mimeType, namaFail);
  const failBaru = folderTahun.createFile(blob);
  
  const urlFail = failBaru.getUrl();


  
  // 4. SIMPAN URL SIJIL KE DALAM EXCEL (Lajur O / 15)
  sheet.getRange(rowSheet, 15).setValue(urlFail); 
  
  return jsonResponse({ 
    status: "BERJAYA", 
    mesej: `Sijil berjaya dimuat naik ke dalam folder Tahun ${tahunKPI}.`, 
    url: urlFail 
  });
}

function muatNaikSijilPenyelaras(data, ss) {
  try {
    // Wajib wujudkan tab "SIJIL_PENYELARAS" di Google Sheets anda 
    let sheet = ss.getSheetByName("SIJIL_PENYELARAS");
    if (!sheet) {
      sheet = ss.insertSheet("SIJIL_PENYELARAS"); 
      sheet.appendRow(["Tahun", "Jabatan", "Nama", "No IC", "Peranan", "URL Sijil"]);
    }

    const tahun = data.tahun;
    const jabatan = data.jabatan;
    const namaPenyelaras = data.namaPenyelaras || "-";
    const icPenyelaras = data.icPenyelaras || "-"; // INI ADALAH KUNCI UTAMA (PRIMARY KEY)
    const perananPenyelaras = data.perananPenyelaras || "-";
    
    const failBase64 = data.failBase64; 
    const namaFail = data.namaFail;
    const mimeType = data.mimeType;


// ---------------------------------------------------------
    // 1. TETAPKAN FORMAT NAMA FAIL: NO_IC_JABATAN_NAMA
    // ---------------------------------------------------------
    let ekstensi = "";
    // 🌟 KEMAS KINI: Tukar namaFailAsal kepada namaFail
    if (namaFail && namaFail.includes(".")) {

    }
    const namaFailStandard = icPenyelaras + "_" + jabatan + "_" + namaPenyelaras + ekstensi;


    /// ---------------------------------------------------------
    // 2. Pengurusan Struktur Folder Bertingkat (Tahun -> Jabatan)
    // ---------------------------------------------------------
    const folderUtama = DriveApp.getFolderById(FOLDER_ID_SIJILKPI);
    
    let folderTahun;
    const subFoldersTahun = folderUtama.getFoldersByName(tahun);
    if (subFoldersTahun.hasNext()) { 
      folderTahun = subFoldersTahun.next(); 
    } else { 
      folderTahun = folderUtama.createFolder(tahun); 
    }

    let folderJabatan;
    const subFoldersJabatan = folderTahun.getFoldersByName(jabatan.toUpperCase());
    if (subFoldersJabatan.hasNext()) { 
      folderJabatan = subFoldersJabatan.next(); 
    } else { 
      folderJabatan = folderTahun.createFolder(jabatan.toUpperCase()); 
    }

    // ---------------------------------------------------------
    // 3. 🌟 SEMAK DAN PADAM FAIL LAMA DENGAN NAMA YANG SAMA DI DRIVE 🌟
    // ---------------------------------------------------------
    const failLama = folderJabatan.getFilesByName(namaFailStandard);
    while (failLama.hasNext()) {
      failLama.next().setTrashed(true);
    }

    // ---------------------------------------------------------
    // Pembersihan Kod Base64 & Upload
    // ---------------------------------------------------------
    let teksBersih = failBase64;
    if (failBase64.indexOf(",") !== -1) { teksBersih = failBase64.split(",")[1]; }
    
    const decodedData = Utilities.base64Decode(teksBersih); 
    // 🌟 KEMAS KINI PENTING: Guna namaFailStandard supaya nama fail ditukar dengan betul!
    const blob = Utilities.newBlob(decodedData, mimeType, namaFailStandard);
    const failBaru = folderJabatan.createFile(blob);
    
    const urlFail = failBaru.getUrl();
    
    // ---------------------------------------------------------
    // Pengemaskinian / Penambahan Rekod di Google Sheets (Kemaskini: Cari IC)
    // ---------------------------------------------------------
    const semuaData = sheet.getDataRange().getValues();
    let rowSheet = -1;
    
    for (let i = 1; i < semuaData.length; i++) {
      
      // 🌟 KEMASKINI: Sekarang kita check Tahun, Jabatan, DAN NO IC (Index 3 / Lajur D)
      let rowTahun = String(semuaData[i][0]).trim();
      let rowJabatan = String(semuaData[i][1]).trim().toUpperCase();
      let rowIC = String(semuaData[i][3]).trim();

      if (rowTahun === String(tahun).trim() && rowJabatan === String(jabatan).trim().toUpperCase() && rowIC === String(icPenyelaras).trim()) {
        rowSheet = i + 1;

        // 🌟 DOUBLE CHECK: Padam URL Sijil lama dari rekod Sheet (jika wujud)
        let urlLama = String(semuaData[i][5]).trim();
        if (urlLama !== "") {
          try {
            let fileIdMatch = urlLama.match(/[-\w]{25,}/);
            if (fileIdMatch && fileIdMatch[0]) {
              DriveApp.getFileById(fileIdMatch[0]).setTrashed(true);
            }
          } catch(e) {} // Abaikan jika fail dah dipadam di Langkah 3
        }




        break;
      }
    }

    if (rowSheet !== -1) {
      // Jika rekod wujud untuk IC tersebut: Kemaskini Nama, IC, Peranan dan URL
      sheet.getRange(rowSheet, 3, 1, 4).setValues([[namaPenyelaras, icPenyelaras, perananPenyelaras, urlFail]]); 
    } else {
      // Jika rekod baharu (IC tak pernah upload sijil untuk Tahun/Jabatan tu): Masukkan baris baru
      sheet.appendRow([tahun, jabatan.toUpperCase(), namaPenyelaras, icPenyelaras, perananPenyelaras, urlFail]);
    }
    
    return ContentService.createTextOutput(JSON.stringify({ 
      status: "BERJAYA", 
      mesej: "Sijil Penyelaras berjaya dimuat naik.", 
      url: urlFail 
    })).setMimeType(ContentService.MimeType.JSON);

  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({ 
      status: "GAGAL", 
      mesej: error.toString() 
    })).setMimeType(ContentService.MimeType.JSON);
  }
}






function prosesPadamSijilPenyelaras(data) {
  try {
    var sheetName = "SIJIL_PENYELARAS"; 
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(sheetName);
    
    // Jika sheet tiada
    if (!sheet) {
      return ContentService.createTextOutput(JSON.stringify({
        status: "GAGAL", 
        mesej: "Sheet " + sheetName + " tidak dijumpai."
      })).setMimeType(ContentService.MimeType.JSON);
    }

    var sheetData = sheet.getDataRange().getValues();
    var rowUntukDipadam = -1;
    var urlSijilDrive = ""; // 🌟 KENA TAMBAH INI: Untuk simpan URL Sijil
    // Loop dari bawah ke atas supaya susunan tak lari bila baris dipadam
    for (var i = sheetData.length - 1; i >= 1; i--) { 
      
      // Susunan lajur berdasarkan SHEET SIJIL_PENYELARAS:
      // [0]=Tahun, [1]=Jabatan, [2]=Nama, [3]=No IC, [4]=Peranan, [5]=URL Sijil
      var rowTahun = String(sheetData[i][0]).trim();    
      var rowJabatan = String(sheetData[i][1]).trim();  
      var rowIC = String(sheetData[i][3]).trim();       

      // Semak jika IC, Jabatan dan Tahun adalah sama
      if (rowIC === String(data.icPenyelaras).trim() && 
          rowJabatan.toUpperCase() === String(data.jabatan).trim().toUpperCase() && 
          rowTahun === String(data.tahun).trim()) {
        
        rowUntukDipadam = i + 1; // Tambah 1 kerana index array mula dari 0, baris sheet mula dari 1
        urlSijilDrive = String(sheetData[i][5]).trim(); // 🌟 KENA TAMBAH INI: Tangkap URL dari Lajur F (Index 5)
        break; 
      }
    }

    // Jika rekod dijumpai, padam baris tersebut
    if (rowUntukDipadam !== -1) {

      // 🌟 [KOD BARU] Padam fail di Google Drive
      if (urlSijilDrive !== "") {
        try {
          // Guna Regex untuk ekstrak ID Fail dari URL yang panjang
          var fileIdMatch = urlSijilDrive.match(/[-\w]{25,}/); 
          if (fileIdMatch && fileIdMatch[0]) {
            var idFail = fileIdMatch[0];
            // setTrashed(true) akan buang fail ke tong sampah (Bin) Drive
            DriveApp.getFileById(idFail).setTrashed(true); 
          }
        } catch (driveError) {
          // Tangkap error jika fail tiada/sudah dipadam di Drive tapi teruskan padam row di Sheet
          console.log("Ralat memadam fail di Drive: " + driveError.toString());
        }
      }

      sheet.deleteRow(rowUntukDipadam);
      
      return ContentService.createTextOutput(JSON.stringify({
        status: "BERJAYA", 
        mesej: "Rekod dan fail sijil telah berjaya dipadam sepenuhnya."
      })).setMimeType(ContentService.MimeType.JSON);
      
    } else {
      return ContentService.createTextOutput(JSON.stringify({
        status: "GAGAL", 
        mesej: "Rekod sijil tidak dijumpai di dalam pangkalan data."
      })).setMimeType(ContentService.MimeType.JSON);
    }
 } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({
      status: "GAGAL", 
      mesej: "Ralat Server: " + error.toString()
    })).setMimeType(ContentService.MimeType.JSON);
  }
}

// 4. LOGIK MUAT NAIK FAIL & AUTO-PADAM
// ==========================================
function kemaskiniTemplateBorang(data) {
  try {
    var folderId = data.FOLDER_ID;
    var fileName = data.NAMA_FAIL;
    var fileData = data.FAIL.dataBase64;
    var mimeType = data.FAIL.mimeType;

    var folder;
    try {
      folder = DriveApp.getFolderById(folderId);
    } catch (e) {
      throw new Error("Gagal akses folder. Pastikan ID Folder betul. Info: " + e.message);
    }

    // --- SISTEM AUTO-DELETE BERMULA DI SINI ---
    // Cari fail lama dengan nama yang sama
    var failLama = folder.getFilesByName(fileName);
    while (failLama.hasNext()) {
      var f = failLama.next();
      try {
        // Guna Advanced API untuk DELETE terus fail lama secara automatik!
        Drive.Files.remove(f.getId(), {supportsAllDrives: true}); 
      } catch (e) {
        // (Backup/Pelan B) Jika atas sebab tertentu Google halang delete, 
        // kita tukar nama supaya fail baharu tetap boleh masuk tanpa ralat.
        var masaTerkini = new Date().getTime();
        f.setName("LAMA_" + masaTerkini + "_" + fileName); 
      }
    }
    // --- SISTEM AUTO-DELETE TAMAT ---

    // Bina fail baharu dari PC
    var decodedData = Utilities.base64Decode(fileData);
    var blob = Utilities.newBlob(decodedData, mimeType, fileName);
    var newFile = folder.createFile(blob);

    // Kosongkan Cache supaya butang muat turun terus di-update
    var cache = CacheService.getScriptCache();
    cache.remove("CACHE_BORANG_" + folderId);

    return {
      status: "BERJAYA",
      idBaru: newFile.getId(),
      mesej: "Fail berjaya dimuat naik dan fail lama telah dipadam secara automatik."
    };

  } catch (err) {
    return {
      status: "GAGAL",
      mesej: "Ralat spesifik: " + err.message
    };
  }
}

// 5. LOGIK AMBIL PAUTAN DINAMIK DENGAN CACHE
// ==========================================
function ambilPautanBorang(folderId) {
  var cache = CacheService.getScriptCache();
  var cacheKey = "CACHE_BORANG_" + folderId;
  
  var cachedData = cache.get(cacheKey);
  if (cachedData) {
    return JSON.parse(cachedData); 
  }

  var folder = DriveApp.getFolderById(folderId);
  var files = folder.getFiles();
  var senaraiFail = {};

  while (files.hasNext()) {
    var file = files.next();
    senaraiFail[file.getName()] = "https://drive.google.com/uc?export=download&id=" + file.getId();
  }

  var hasil = {
    status: "BERJAYA",
    data: senaraiFail
  };

  cache.put(cacheKey, JSON.stringify(hasil), 21600);

  return hasil;
}

// FUNGSI UPLOAD SIJIL KE GOOGLE DRIVE & UPDATE SHEET
// ========================================================
function fungsiUploadSijil(payload, ss) { 
  try {
    if (!ss) {
      ss = SpreadsheetApp.getActiveSpreadsheet(); 
    }

    var bil = payload.bil;
    var tahun = payload.tahun || "Tiada_Tahun";
    var jabatan = payload.jabatan || "Tiada_Jabatan";
    
    // 1. GUNA FOLDER ID KHAS UNTUK FOLDER UTAMA
    var folderIdUtama = "1BfNhDB6FHnbBs5jOVnRjDKmgZG9PjXsV"; 
    
    // UJI AKSES FOLDER (Jika ralat di sini, bermaksud e-mel skrip tiada akses Editor ke folder ini)
    var parentFolder;
    try {
      parentFolder = DriveApp.getFolderById(folderIdUtama);
    } catch(e) {
      return ContentService.createTextOutput(JSON.stringify({
        status: "GAGAL",
        mesej: "Akses Ditolak: E-mel yang menjalankan skrip ini BUKAN 'Editor' kepada Folder Utama Google Drive tersebut."
      })).setMimeType(ContentService.MimeType.JSON);
    }
    
    // Cari atau cipta subfolder 'Tahun'
    var tahunFolderIterator = parentFolder.getFoldersByName(tahun);
    var tahunFolder = tahunFolderIterator.hasNext() ? tahunFolderIterator.next() : parentFolder.createFolder(tahun);
    
    // Cari atau cipta subfolder 'Jabatan'
    var jabatanFolderIterator = tahunFolder.getFoldersByName(jabatan);
    var jabatanFolder = jabatanFolderIterator.hasNext() ? jabatanFolderIterator.next() : tahunFolder.createFolder(jabatan);
    
    // 2. BINA & SIMPAN FAIL BARU
    var decodedData = Utilities.base64Decode(payload.data);
    var blob = Utilities.newBlob(decodedData, payload.mimeType, payload.namaFail);
    var file = jabatanFolder.createFile(blob);
    
    // 🌟 KOD BARU: Abaikan ralat jika pemilik folder sekat penukaran Permission
    try {
      file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
    } catch(shareErr) {
      // Abaikan. Fail akan ikut permission folder secara automatik.
    }
    
    var fileUrl = file.getUrl();
    
    // 3. CARI REKOD DI GOOGLE SHEET DAN UPDATE LINK SERTA PADAM FAIL LAMA
    var sheet = ss.getSheetByName("INOVASI"); 
    
    if (!sheet) {
        return ContentService.createTextOutput(JSON.stringify({
          status: "GAGAL",
          mesej: "Ralat: Tab 'INOVASI' tidak dijumpai di dalam Google Sheet."
        })).setMimeType(ContentService.MimeType.JSON);
    }

    var data = sheet.getDataRange().getDisplayValues(); 
    var dijumpai = false;
    
    for (var i = 1; i < data.length; i++) {
      var sheetBil = String(data[i][0]).trim();
      var payloadBil = String(bil).trim();
      
      if (sheetBil === payloadBil && sheetBil !== "") {
        
        var urlSijilLama = String(data[i][10]).trim(); 
        
        if (urlSijilLama !== "") {
          try {
            var fileIdMatch = urlSijilLama.match(/[-\w]{25,}/);
            if (fileIdMatch && fileIdMatch[0]) {
              DriveApp.getFileById(fileIdMatch[0]).setTrashed(true);
            }
          } catch (e) {
            // Abaikan jika fail tak jumpa atau tiada kebenaran padam
          }
        }

        sheet.getRange(i + 1, 11).setValue(fileUrl);
        dijumpai = true;
        break;
      }
    }
    
    // JIKA TIDAK JUMPA REKOD
    if (!dijumpai) {
        try { file.setTrashed(true); } catch(e){} // Padam fail baru jika error
        return ContentService.createTextOutput(JSON.stringify({
          status: "GAGAL",
          mesej: "Rekod tidak dijumpai! Sistem mencari BIL: '" + bil + "' di Lajur A."
        })).setMimeType(ContentService.MimeType.JSON);
    }
    
    return ContentService.createTextOutput(JSON.stringify({
      status: "BERJAYA",
      mesej: "Sijil berjaya dikemaskini.",
      url: fileUrl
    })).setMimeType(ContentService.MimeType.JSON);
    
  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({
      status: "GAGAL",
      mesej: "Ralat sistem: " + error.message
    })).setMimeType(ContentService.MimeType.JSON);
  }
}
